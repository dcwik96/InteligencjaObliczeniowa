	// --- my Tank -------------
	var me = e.data.myTank;
	var myCtrl = me.controls;


	// --- enemy Tank ----------
	var enemy = e.data.enemyTank;

	// --- current game Time ---
	var cgt = e.data.currentGameTime;

	var input = [me.x/500,
					me.y/500,

					me.shootCooldown/100,
					enemy.x/500,
					enemy.y/500,
					e.data.currentGameTime/40000
				];

	var weights1 = [-0.847886847134174, -1.69009844513508, 3.06082122429591, -2.05805572880234, 0.809794106538265, 0.0445363817138832, -1.70003623705589, -1.31212426358934, -1.85906380427056, -0.873337728989258, 0.639547584631172, -1.07532897238582, -0.318821131529731, -0.899273884722628, 2.2444870795631, 0.333932778767224, 0.402384948980428, -0.363276866287009, 0.0104052665773477, -0.103158796317718, 0.491342451426235, 0.255418201458929, -2.18410165377388, -1.24493598306807, -0.939256645393268, -2.53364383615537, -0.565125759473863, -0.774293306778966, -2.06181715891916, -0.707202235714628, -1.09839620050768, -0.790318649767228, -0.360932941048802, -1.26113428214927, -1.00315666162677];
	var bias1 = [1.52931647304191, -0.683762815022971, -1.26947564545938, 0.0648237748558423, 0.527393151442628];
	var hidden = [0, 0, 0, 0, 0];

	var i, j;
	for (i = 0; i < 5; i++) {
		for (j = 0; j < 7; j++) {
			hidden[i] += input[j] * weights1[i * 7 + j];
		}
		hidden[i] = hidden[i] + bias1[i];
		hidden[i] = 1 / (1 + Math.pow(Math.E, -hidden[i]));
	}
	//~ console.log("[hidden]: " + hidden);

	var weights2 = [-0.0715219724498689, 0.656687898123871, -1.11701070787621, 0.0510900443252304, -1.28929716883776, -0.0787577505273848, -1.32706630039177, 0.921111283344343, 0.599225576139466, -0.293513662563517, -0.601571880151595, 2.97021718898804, 0.344393829236086, 0.106479198779836, 0.243149041412368, -0.984868534275229, -0.664241851521269, -0.777490857196657, -0.192908636340889, -0.00261295932656624];

		for (j = 0; j < 5; j++) {
			output[i] += hidden[j] * weights2[i * 5 + j];
		}
		output[i] = output[i] + bias2[i];
		output[i] = 1 / (1 + Math.pow(Math.E, -output[i]));
		output[i] = Math.round(output[i]);
	}

	response.turnLeft = output[0];
	response.turnRight = output[1];
	response.goForward = output[2];
	response.shoot = output[3];

	self.postMessage(response);
}
